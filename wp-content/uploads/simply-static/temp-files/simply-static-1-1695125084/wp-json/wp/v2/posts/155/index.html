(D\u00e2n tr\u00ed) – Nhi\u1ec1u b\u1ec7nh vi\u1ec7n v\u00e0 trung t\u00e2m y t\u1ebf \u0111\u00e3 tr\u1edf th\u00e0nh m\u1ee5c ti\u00eau t\u1ea5n c\u00f4ng c\u1ee7a tin t\u1eb7c trong th\u1eddi gian qua, \u0111i\u1ec1u n\u00e0y khi\u1ebfn t\u00ednh m\u1ea1ng c\u1ee7a b\u1ec7nh nh\u00e2n tr\u1ef1c ti\u1ebfp b\u1ecb \u0111e d\u1ecda.<\/h2>\n
\n

H\u1ec7 th\u1ed1ng c\u1ea5p c\u1ee9u t\u00ea li\u1ec7t ho\u1eb7c b\u1ecb gi\u00e1n \u0111o\u1ea1n<\/strong><\/p>\n

Nh\u1eafc \u0111\u1ebfn tin t\u1eb7c, nhi\u1ec1u ng\u01b0\u1eddi ch\u1ec9 ngh\u0129 \u0111\u1ebfn nh\u1eefng tin t\u1eb7c t\u1ea5n c\u00f4ng h\u1ec7 th\u1ed1ng m\u1ea1ng, h\u1ec7 th\u1ed1ng ph\u1ea7n m\u1ec1m \u0111\u1ec3 l\u1ea5y c\u1eafp th\u00f4ng tin nh\u1eb1m ki\u1ebfm l\u1ee3i nhu\u1eadn ho\u1eb7c m\u1ee5c \u0111\u00edch ch\u00ednh tr\u1ecb.<\/p>\n

Nh\u01b0ng gi\u1edd \u0111\u00e2y, tin t\u1eb7c c\u00f3 th\u1ec3 tr\u1edf th\u00e0nh nh\u1eefng s\u00e1t nh\u00e2n theo \u0111\u00fang ngh\u0129a \u0111en khi ch\u00fang chuy\u1ec3n h\u01b0\u1edbng t\u1ea5n c\u00f4ng v\u00e0o h\u1ec7 th\u1ed1ng b\u1ec7nh vi\u1ec7n v\u00e0 c\u00e1c trung t\u00e2m ch\u0103m s\u00f3c s\u1ee9c kh\u1ecfe.<\/p>\n

Cu\u1ed1i tu\u1ea7n tr\u01b0\u1edbc, hacker \u0111\u00e3 th\u1ef1c hi\u1ec7n h\u00e0ng lo\u1ea1t v\u1ee5 t\u1ea5n c\u00f4ng quy m\u00f4 l\u1edbn nh\u1eb1m v\u00e0o h\u1ec7 th\u1ed1ng m\u00e1y t\u00ednh c\u1ee7a Prospect Medical Holdings, m\u1ed9t nh\u00e0 cung c\u1ea5p d\u1ecbch v\u1ee5 ch\u0103m s\u00f3c s\u1ee9c kh\u1ecfe c\u00f3 tr\u1ee5 s\u1edf t\u1ea1i bang California (M\u1ef9) v\u00e0 c\u00f3 h\u1ec7 th\u1ed1ng b\u1ec7nh vi\u1ec7n t\u1ea1i c\u00e1c bang Connecticut, Pennsylvania, Rhode Island v\u00e0 Texas.
\nC\u00e1c cu\u1ed9c t\u1ea5n c\u00f4ng m\u1ea1ng nh\u1eb1m v\u00e0o b\u1ec7nh vi\u1ec7n, trung t\u00e2m y t\u1ebf\u2026 s\u1ebd g\u00e2y n\u00ean h\u1eadu qu\u1ea3 nghi\u00eam tr\u1ecdng v\u00e0 \u1ea3nh h\u01b0\u1edfng \u0111\u1ebfn t\u00ednh m\u1ea1ng b\u1ec7nh nh\u00e2n (\u1ea2nh minh h\u1ecda: Pinterest).<\/p>\n

V\u1ee5 t\u1ea5n c\u00f4ng m\u1ea1ng \u0111\u00e3 l\u00e0m h\u1ec7 th\u1ed1ng m\u00e1y t\u00ednh t\u1ea1i nhi\u1ec1u b\u1ec7nh vi\u1ec7n b\u1ecb \u1ea3nh h\u01b0\u1edfng, khi\u1ebfn ph\u00f2ng c\u1ea5p c\u1ee9u t\u1ea1i m\u1ed9t s\u1ed1 b\u1ec7nh vi\u1ec7n c\u1ee7a Prospect Medical Holdings ph\u1ea3i \u0111\u00f3ng c\u1eeda, trung t\u00e2m \u0111i\u1ec1u h\u00e0nh xe c\u1ee9u th\u01b0\u01a1ng b\u1ecb gi\u00e1n \u0111o\u1ea1n khi\u1ebfn xe kh\u00f4ng th\u1ec3 v\u1eadn chuy\u1ec3n ng\u01b0\u1eddi b\u1ec7nh \u0111\u1ebfn c\u00e1c trung t\u00e2m y t\u1ebf\u2026<\/p>\n

M\u1ed9t s\u1ed1 trung t\u00e2m ph\u1eabu thu\u1eadt v\u00e0 khoa kh\u00e1m b\u1ec7nh t\u1ea1i c\u00e1c b\u1ec7nh vi\u1ec7n b\u1ecb t\u1ea5n c\u00f4ng m\u1ea1ng c\u0169ng ch\u1ecbu \u1ea3nh h\u01b0\u1edfng v\u00e0 kh\u00f4ng th\u1ec3 ho\u1ea1t \u0111\u1ed9ng ho\u1eb7c ch\u1ec9 ho\u1ea1t \u0111\u1ed9ng c\u1ea7m ch\u1eebng.<\/p>\n

Nhi\u1ec1u b\u1ec7nh vi\u1ec7n b\u1ecb tin t\u1eb7c t\u1ea5n c\u00f4ng \u0111\u00e3 ph\u1ea3i chuy\u1ec3n sang s\u1eed d\u1ee5ng gi\u1ea5y t\u1edd v\u00e0 qu\u1ea3n l\u00fd h\u1ed3 s\u01a1 b\u1ec7nh nh\u00e2n theo c\u00e1ch truy\u1ec1n th\u1ed1ng khi c\u00e1c h\u1ec7 th\u1ed1ng m\u00e1y t\u00ednh \u0111\u1ec1u b\u1ecb ng\u01b0ng ho\u1ea1t \u0111\u1ed9ng ho\u1eb7c b\u1ecb ng\u1eaft k\u1ebft n\u1ed1i m\u1ea1ng \u0111\u1ec3 tr\u00e1nh hacker x\u00e2m nh\u1eadp.<\/p>\n

\u0110\u1ebfn th\u1eddi \u0111i\u1ec3m hi\u1ec7n t\u1ea1i, m\u1ed9t s\u1ed1 b\u1ec7nh vi\u1ec7n \u0111\u00e3 c\u00f3 th\u1ec3 ho\u1ea1t \u0111\u1ed9ng b\u00ecnh th\u01b0\u1eddng tr\u1edf l\u1ea1i, nh\u01b0ng m\u1ed9t s\u1ed1 b\u1ec7nh vi\u1ec7n thu\u1ed9c quy\u1ec1n qu\u1ea3n l\u00fd c\u1ee7a Prospect Medical Holdings s\u1ebd ph\u1ea3i ng\u1eebng ho\u1ea1t \u0111\u1ed9ng \u00edt nh\u1ea5t \u0111\u1ebfn tu\u1ea7n sau \u0111\u1ec3 kh\u1eafc ph\u1ee5c h\u1eadu qu\u1ea3 c\u1ee7a v\u1ee5 t\u1ea5n c\u00f4ng m\u1ea1ng.<\/p>\n

Vi\u1ec7c c\u00e1c ph\u00f2ng c\u1ea5p c\u1ee9u v\u00e0 h\u1ec7 th\u1ed1ng \u0111i\u1ec1u chuy\u1ec3n xe c\u1ee9u th\u01b0\u01a1ng b\u1ecb gi\u00e1n \u0111o\u1ea1n \u0111\u00e3 khi\u1ebfn t\u00ednh m\u1ea1ng c\u1ee7a nhi\u1ec1u b\u1ec7nh nh\u00e2n b\u1ecb \u0111e d\u1ecda. Hi\u1ec7n ch\u01b0a r\u00f5 c\u00f3 thi\u1ec7t h\u1ea1i n\u00e0o v\u1ec1 ng\u01b0\u1eddi sau v\u1ee5 t\u1ea5n c\u00f4ng m\u1ea1ng n\u00e0y hay kh\u00f4ng.<\/p>\n

C\u1ee5c \u0111i\u1ec1u tra Li\u00ean bang (FBI) v\u00e0 c\u00e1c chuy\u00ean gia an ninh m\u1ea1ng \u0111\u00e3 v\u00e0o cu\u1ed9c \u0111\u1ec3 \u0111i\u1ec1u tra th\u1ee7 ph\u1ea1m v\u1ee5 t\u1ea5n c\u00f4ng m\u1ea1ng nh\u1eb1m v\u00e0o c\u00e1c ph\u00f2ng c\u1ea5p c\u1ee9u v\u00e0 b\u1ec7nh vi\u1ec7n t\u1ea1i nhi\u1ec1u bang c\u1ee7a M\u1ef9.<\/p>\n

“Khi v\u1ee5 t\u1ea5n c\u00f4ng m\u1ea1ng x\u1ea3y ra, ch\u00fang t\u00f4i \u0111\u00e3 ph\u1ea3i t\u1eaft \u0111i h\u1ec7 th\u1ed1ng c\u1ee7a m\u00ecnh \u0111\u1ec3 gi\u1ea3m h\u1eadu qu\u1ea3. Ch\u00fang t\u00f4i \u0111ang h\u1ee3p t\u00e1c v\u1edbi FBI v\u00e0 c\u00e1c chuy\u00ean gia an ninh m\u1ea1ng \u0111\u1ec3 \u0111i\u1ec1u tra v\u1ee5 t\u1ea5n c\u00f4ng v\u00e0 x\u00e1c \u0111\u1ecbnh th\u1ee7 ph\u1ea1m \u0111\u1ee9ng sau v\u1ee5 vi\u1ec7c”, \u0111\u1ea1i di\u1ec7n c\u1ee7a Prospect Medical Holdings chia s\u1ebb.<\/p>\n

B\u1ec7nh vi\u1ec7n tr\u1edf th\u00e0nh m\u1ee5c ti\u00eau m\u1edbi c\u1ee7a nh\u1eefng k\u1ebb s\u00e1t nh\u00e2n g\u00f5 ph\u00edm<\/strong><\/p>\n

Theo c\u00e1c chuy\u00ean gia b\u1ea3o m\u1eadt, b\u1ec7nh vi\u1ec7n v\u00e0 c\u00e1c trung t\u00e2m ch\u0103m s\u00f3c s\u1ee9c kh\u1ecfe tr\u00ean to\u00e0n c\u1ea7u \u0111ang tr\u1edf th\u00e0nh m\u1ee5c ti\u00eau t\u1ea5n c\u00f4ng c\u1ee7a tin t\u1eb7c trong nh\u1eefng n\u0103m qua.<\/p>\n

Theo k\u1ebft qu\u1ea3 m\u1ed9t cu\u1ed9c kh\u1ea3o s\u00e1t \u0111\u01b0\u1ee3c c\u00f4ng b\u1ed1 c\u1ee7a c\u00f4ng ty an ninh m\u1ea1ng Censinet (M\u1ef9), h\u01a1n 1\/4 trung t\u00e2m ch\u0103m s\u00f3c s\u1ee9c kh\u1ecfe t\u1ea1i M\u1ef9 \u0111\u00e3 b\u1ecb t\u1ea5n c\u00f4ng m\u1ea1ng trong 2 n\u0103m qua, ch\u1ee7 y\u1ebfu b\u1edfi c\u00e1c lo\u1ea1i m\u00e3 \u0111\u1ed9c t\u1ed1ng ti\u1ec1n, khi\u1ebfn t\u1ef7 l\u1ec7 b\u1ec7nh nh\u00e2n t\u1eed vong t\u0103ng cao v\u00e0 \u1ea3nh h\u01b0\u1edfng qu\u00e1 tr\u00ecnh ch\u0103m s\u00f3c s\u1ee9c kh\u1ecfe ng\u01b0\u1eddi b\u1ec7nh.<\/p>\n

K\u1ebft qu\u1ea3 nghi\u00ean c\u1ee9u c\u1ee7a Censinet cho bi\u1ebft c\u00e1c v\u1ee5 t\u1ea5n c\u00f4ng m\u1ea1ng g\u00e2y ra h\u1eadu qu\u1ea3 c\u00f2n nghi\u00eam tr\u1ecdng h\u01a1n bi\u1ebfn ch\u1ee9ng t\u1eeb c\u00e1c th\u1ee7 thu\u1eadt y t\u1ebf v\u00e0 22% c\u00e1c t\u1ed5 ch\u1ee9c y t\u1ebf tham gia kh\u1ea3o s\u00e1t cho bi\u1ebft s\u1ed1 ca b\u1ec7nh t\u1eed vong \u0111\u00e3 t\u0103ng l\u00ean do c\u00e1c cu\u1ed9c t\u1ea5n c\u00f4ng m\u1ea1ng.<\/p>\n

K\u1ebft qu\u1ea3 nghi\u00ean c\u1ee9u c\u1ee7a Censinet cho th\u1ea5y nh\u1eefng cu\u1ed9c t\u1ea5n c\u00f4ng m\u1ea1ng c\u1ee7a tin t\u1eb7c kh\u00f4ng ch\u1ec9 \u0111\u01a1n thu\u1ea7n g\u00e2y ra thi\u1ec7t h\u1ea1i v\u1ec1 t\u00e0i ch\u00ednh, d\u1eef li\u1ec7u c\u00e1 nh\u00e2n ho\u1eb7c c\u00e1c d\u1ecbch v\u1ee5\u2026 m\u00e0 c\u00f2n \u1ea3nh h\u01b0\u1edfng \u0111\u1ebfn c\u1ea3 s\u1ee9c kh\u1ecfe v\u00e0 t\u00ednh m\u1ea1ng con ng\u01b0\u1eddi.<\/p>\n

Cu\u1ed1i n\u0103m 2021, m\u1ed9t n\u1eef b\u1ec7nh nh\u00e2n ng\u01b0\u1eddi \u0110\u1ee9c \u0111\u00e3 thi\u1ec7t m\u1ea1ng khi h\u1ec7 th\u1ed1ng m\u00e1y t\u00ednh c\u1ee7a b\u1ec7nh vi\u1ec7n b\u1ecb nhi\u1ec5m m\u00e3 \u0111\u1ed9c t\u1ed1ng ti\u1ec1n, khi\u1ebfn to\u00e0n b\u1ed9 h\u1ec7 th\u1ed1ng b\u1ecb ng\u01b0ng tr\u1ec7 l\u00e0m \u1ea3nh h\u01b0\u1edfng \u0111\u1ebfn qu\u00e1 tr\u00ecnh \u0111i\u1ec1u tr\u1ecb c\u1ee7a b\u1ec7nh nh\u00e2n x\u1ea5u s\u1ed1 n\u00e0y.<\/p>\n

C\u00e1c chuy\u00ean gia b\u1ea3o m\u1eadt \u0111\u00e3 nhi\u1ec1u l\u1ea7n c\u1ea3nh b\u00e1o c\u00e1c b\u1ec7nh vi\u1ec7n, trung t\u00e2m y t\u1ebf s\u1ebd tr\u1edf th\u00e0nh m\u1ee5c ti\u00eau t\u1ea5n c\u00f4ng c\u1ee7a tin t\u1eb7c v\u00e0 c\u00e1c lo\u1ea1i m\u00e3 \u0111\u1ed9c, nh\u01b0ng cho \u0111\u1ebfn nay nh\u1eefng \u0111\u1ecba \u0111i\u1ec3m n\u00e0y v\u1eabn ch\u01b0a c\u00f3 s\u1ef1 chu\u1ea9n b\u1ecb th\u1ef1c s\u1ef1 \u0111\u1ea7y \u0111\u1ee7.<\/p>\n

Nhi\u1ec1u thi\u1ebft b\u1ecb kh\u00e1m ch\u1eefa b\u1ec7nh hi\u1ec7n \u0111\u1ea1i \u0111\u00f2i h\u1ecfi k\u1ebft n\u1ed1i internet \u0111\u1ec3 s\u1eed d\u1ee5ng v\u00e0 n\u1ebfu b\u1ecb m\u00e3 \u0111\u1ed9c t\u1ea5n c\u00f4ng, c\u00e1c thi\u1ebft b\u1ecb n\u00e0y kh\u00f4ng th\u1ec3 ho\u1ea1t \u0111\u1ed9ng s\u1ebd l\u00e0m \u1ea3nh h\u01b0\u1edfng nghi\u00eam tr\u1ecdng \u0111\u1ebfn qu\u00e1 tr\u00ecnh ch\u1eefa tr\u1ecb cho b\u1ec7nh nh\u00e2n.<\/p>\n

Ngay c\u1ea3 khi c\u00e1c v\u1ee5 t\u1ea5n c\u00f4ng m\u1ea1ng kh\u00f4ng nh\u1eb1m v\u00e0o h\u1ec7 th\u1ed1ng m\u00e1y m\u00f3c, thi\u1ebft b\u1ecb y t\u1ebf c\u1ee7a b\u1ec7nh vi\u1ec7n m\u00e0 nh\u1eb1m m\u1ee5c ti\u00eau l\u1ea5y c\u1eafp d\u1eef li\u1ec7u c\u1ee7a ng\u01b0\u1eddi b\u1ec7nh, \u0111i\u1ec1u n\u00e0y c\u0169ng c\u00f3 th\u1ec3 \u1ea3nh h\u01b0\u1edfng \u0111\u1ebfn k\u1ebft qu\u1ea3 ch\u1eefa tr\u1ecb c\u1ee7a b\u1ec7nh nh\u00e2n.<\/p>\n
